factorial

#include<stdio.h>
int main(){
    int n=10;
    int fact=1;
    for(int i=1;i<=n;i++){
       fact=fact*i; 
    }
    printf("%d ",fact);
    return 0;
}


Fibonacci series

#include<stdio.h>
int main(){
    int n1=0,n2=1,n3;
    int n=5;
    printf("%d %d ",n1,n2);
    for(int i=1;i<=n;i++){
        n3=n1+n2;
        printf("%d ",n3);
        n1=n2;
        n2=n3;
    }
    return 0;
}
