#include <stdio.h>

int main() {
    int number;

    // Input number from user
    printf("Enter a number: ");
    scanf("%d", &number);

    // Check if the number is greater than 50
    if (number > 50) {
        printf("The number is greater than 50.\n");
    } else {
        printf("The number is not greater than 50.\n");
    }

    return 0;
}
